        \operation[0]{getMoment(iSphere : int) : double[3]}
    \end{abstractclass}
    
    \begin{class}[text width=\middleTextWidth]{ApolarSpheres}
    {\AbstractSpheresX, \AbstractSpheresY-4}
        \inherit{AbstractSpheres}
        \operation{getMoment(iSphere : int) : double[3]}
    \end{class}
    
    \begin{class}[text width=\middleTextWidth]{DipolarSpheres}
